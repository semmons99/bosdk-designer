      @designer = WIN32OLE.new('Designer.Application')

        args[:authtype] || 'secEnterprise'
      )

    def method_missing(method, *args)
      ole_method_name = method.to_s.sub(/=$/, '')
      if @designer.ole_respond_to?(ole_method_name)
        @designer.send(method, *args)
      else
        super
      end
    end
